#include <iostream>
#include <cmath>
using namespace std;
int main()
{
	double x, y, A, B;
	cout << "x="; cin >> x;
	A = 4.95 * x * x;
	if (x < -3.5)
		B = 4 + pow(x,-2);
	if (x >= -3.5 && x < 1)
		B = tan((3.5 + x) / 5);
	if (x >= 1)
		B = sin(3 * x) - cos(x);
	y = A + B;
	cout << endl;
	cout << "1) y=" << y << endl;
	// 1 спосіб в скороченій;
	if (x < -3.5)
		B = 4 + pow(x,-2);
	else
		if (x >= 1)
			B = sin(3 * x) - cos(x);
		else
			B = tan((3.5 + x) / 5);
	y = A + B;
	cout << "2) y=" << y << endl;
	cin.get();
	return 0;
}
