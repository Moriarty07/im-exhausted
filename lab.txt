#include <iostream>
#include <cmath>
using namespace std;
int main()
{
	double Pi = 4 * atan(1);
	double x, z1, z2;
	cout << "x="; cin >> x;
	z1 = (sin(Pi / 2 + 3 * x)) / (1 - sin(3 * x - Pi));
	z2 = (cos((5 * Pi / 4) + (3 * x / 2))/sin((5 * Pi / 4) + (3 * x / 2)));
	cout << endl;
	cout << "z1=" << z1 << endl;
	cout << "z2=" << z2 << endl;
	cin.get();
	return 0;
}
