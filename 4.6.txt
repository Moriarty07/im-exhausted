#include <iostream>
#include <cmath>
using namespace std;
int main() {
	double S1, S2;
	int n, k;
	k = 1;
	S1 = 0;
	S2 = 0;
	n = 1;
	while (k <= 10)
	{	
		while (n <= k)
		{
			S2 += sin(k*n);
			n++;
		}
		S1 += S2/k;
		k++;
	}
	cout << S1 << endl;
	S1 = 0;
	S2 = 0;
	n = 1;
	k = 1;
	do
	{
		do
		{
			S2 += sin(k * n);
			n++;
		} while (n <= k);
		S1 += S2 / k;
		k++;
	} while (n <= 10);
	cout << S1 << endl;
	
	for (k = 1; k <= 10; k++)
	{
		S1 = 0;
		S2 = 0;
		for (n = 1; n <= k; n++)
		{
			S2 += sin(k * n);
		}
		S1 += S2 / k;
	}
	cout << S1 << endl;
	
	for (k = 10; k >= 1; k--)
	{
		S1 = 0;
		S2 = 0;
		for (n = k; n >= 1; n--)
		{
			S2 += sin(k * n);
		}
		S1 += S2 / k;
	}
	cout << S1 << endl;
	return 0;
}
